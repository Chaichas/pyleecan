28def read_all(path, is_internal=False, in_path=""): 

29 """Read every csv files in a directory and subdirectory and create a structure for the 

30 code generation 

31 

32 Parameters 

33 ---------- 

34 path : str 

35 path to the root folder with the csv files 

36 is_internal : bool 

37 True to overwrite the open source csv files by internal ones 

38 

39 Returns 

40 ------- 

41 gen_dict : dict 

42 Dict with key = class name and value = class dict (name, package, properties, methods...) 

43 """ 

44 gen_dict = dict() 

45 

46 # Read the open source doc 

47 for (dirpath, _, filenames) in walk(path): 

48 for file_name in filenames: 

49 if file_name[-4:] == ".csv" and file_name[:2] != "~$": 

50 # For all .csv file in the folder and subfolder ... 

51 gen_dict[file_name[:-4]] = read_file(join(dirpath, file_name)) 

52 gen_dict[file_name[:-4]]["is_internal"] = False 

53 

54 # Read the Internal doc to adapt the classes (if needed) 

55 if is_internal: 

56 for (dirpath, _, filenames) in walk(in_path): 

57 for file_name in filenames: 

58 if file_name[-4:] == ".csv" and file_name[:2] != "~$": 

59 # For all .csv file in the folder and subfolder ... 

60 print("Using internal version for: " + file_name[:-4]) 

61 gen_dict[file_name[:-4]] = read_file(join(dirpath, file_name)) 

62 gen_dict[file_name[:-4]]["is_internal"] = True 

63 

64 # Update all the "daughters" key according to "mother" key 

65 update_all_daughters(gen_dict) 

66 

67 return gen_dict

70def read_file(path): 

71 """Read a csv file and create a dict for the class code generation 

72 

73 Parameters 

74 ---------- 

75 path : str 

76 path to the class csv file to read 

77 

78 

79 Returns 

80 ------- 

81 class_dict : dict 

82 Dict containing all the class informations (properties, package, methods...) 

83 """ 

84 class_dict = dict() 

85 

86 # Open the module doc 

87 if not isfile(path): 

88 raise NotAFile("File not found") 

89 

90 # The class name is the csv file name 

91 class_dict["name"] = path.split("\\")[-1][:-4] 

92 try: # Cleanup path to avoid "commit noise" 

93 class_dict["path"] = path[path.index(PACKAGE_NAME) :] 

94 except ValueError: # Path doesn't contain pyleecan 

95 class_dict["path"] = path 

96 # Cleanup \ to avoid errors 

97 class_dict["path"] = class_dict["path"].replace("\\", "/") 

98 with open(path, mode="r") as csv_file: 

99 class_csv = reader(csv_file, delimiter=",") 

100 class_csv = list(class_csv) 

101 # Get all the properties of the class 

102 properties = list() 

103 Nline = len(class_csv) 

104 for rx in range(1, Nline): # Skip the first line (title of column) 

105 name = class_csv[rx][NAME_COL] 

106 if name != "": 

107 prop_dict = dict() 

108 prop_dict["name"] = name 

109 prop_dict["unit"] = class_csv[rx][UNIT_COL] 

110 prop_dict["type"] = class_csv[rx][TYPE_COL] 

111 prop_dict["min"] = class_csv[rx][MIN_VAL_COL].replace(",", ".") 

112 prop_dict["max"] = class_csv[rx][MAX_VAL_COL].replace(",", ".") 

113 prop_dict["value"] = class_csv[rx][DEF_VAL_COL] 

114 if prop_dict["type"] == "float": 

115 prop_dict["value"] = prop_dict["value"].replace(",", ".") 

116 if ( 

117 prop_dict["value"] != "" 

118 and prop_dict["type"] != "str" 

119 and "." not in prop_dict["type"] 

120 and "()" not in prop_dict["value"] 

121 ): 

122 prop_dict["value"] = eval(prop_dict["value"]) 

123 desc = class_csv[rx][EN_DESC_COL] 

124 prop_dict["desc"] = desc.replace("\\\\", "\\") 

125 properties.append(prop_dict) 

126 class_dict["properties"] = properties 

127 

128 # Get all the constants 

129 cste = list() 

130 for rx in range(1, Nline): # Skip the first line (title of column) 

131 name = class_csv[rx][CST_NAME_COL] 

132 if name != "": 

133 cste_dict = dict() 

134 cste_dict["name"] = name 

135 cste_dict["value"] = class_csv[rx][CST_VAL_COL] 

136 cste.append(cste_dict) 

137 class_dict["constants"] = cste 

138 

139 # Get all the methods 

140 class_dict["methods"] = list() 

141 for rx in range(1, Nline): # Skip the first line (title of column) 

142 meth = class_csv[rx][METH_COL] 

143 if meth != "": 

144 class_dict["methods"].append(meth) 

145 

146 # The daughters are automatically detected in read_all 

147 class_dict["daughters"] = list() 

148 

149 class_dict["package"] = class_csv[1][PACK_COL] 

150 class_dict["desc"] = class_csv[1][CLASS_DEF_COL] 

151 class_dict["mother"] = class_csv[1][HER_COL] 

152 if class_dict["mother"] == class_dict["name"]: 

153 raise InheritError( 

154 "ERROR: the class " + class_dict["name"] + " inherit from itself" 

155 ) 

156 

157 return class_dict

160def update_all_daughters(gen_dict): 

161 """This function update all the "daughters" key according to the "mother" key 

162 

163 Parameters 

164 ---------- 

165 gen_dict : dict 

166 gen_dict with no daughter set 

167 """ 

168 

169 # list of classes that have a mother 

170 daughter_dict = { 

171 class_name: class_dict 

172 for class_name, class_dict in gen_dict.items() 

173 if class_dict["mother"] not in ["", None] 

174 } 

175 

176 # Update the daughter (sorted to avoid "commit noise") 

177 for name, daughter in iter(sorted(list(daughter_dict.items()))): 

178 # Update the mother 

179 mother = gen_dict[daughter["mother"]] 

180 mother["daughters"].append(name) 

181 # Update all the mother of the mother 

182 while mother["mother"] not in ["", None]: 

183 mother = gen_dict[mother["mother"]] 

184 mother["daughters"].append(name)

187def get_value_str(value, type_val): 

188 """Convert the value from the csv file to the correct str according to the type 

189 

190 Parameters 

191 ---------- 

192 value : str 

193 value to convert 

194 

195 type_val : str 

196 Type to convert to 

197 

198 Returns 

199 ------- 

200 value : str 

201 Value updated to match the type 

202 """ 

203 

204 if value in ["None", None]: 

205 return "None" 

206 elif type_val == "str": 

207 # For str add " " 

208 return '"' + str(value) + '"' 

209 elif type_val == "int": 

210 # For int convert to avoid ".0" 

211 return str(int(value)) 

212 elif type_val == "dict": 

213 return "{}" 

214 elif type_val == "bool": 

215 # change 1 or 0 to True and False 

216 return str(bool(int(value))) 

217 else: 

218 return str(value)

221def find_import_type(gen_dict, class_dict, pyleecan_type=[]): 

222 """Find all the Pyleecan type used by the class of class_dict 

223 

224 Parameters 

225 ---------- 

226 gen_dict : dict 

227 Dict with key = class name and value = class_dict 

228 

229 class_dict : dict 

230 Dict of the class to find the import 

231 

232 pyleecan_type : list 

233 Existing type to import (Default value = []) 

234 

235 Returns 

236 ------- 

237 type_list : list 

238 List of pyleecan type (as str) to import 

239 """ 

240 

241 # Get all properties including mother ones 

242 prop_list = list(class_dict["properties"]) 

243 while class_dict["mother"] != "": 

244 class_dict = gen_dict[class_dict["mother"]] 

245 prop_list.extend(class_dict["properties"]) 

246 

247 # Find every property type of the class 

248 for prop in prop_list: 

249 # Detect list of pyleecan type 

250 if is_list_pyleecan_type(prop["type"]) or is_dict_pyleecan_type(prop["type"]): 

251 prop_type = prop["type"][1:-1] 

252 else: 

253 prop_type = prop["type"] 

254 # Store the non python type once and avoid empty line 

255 if prop_type not in PYTHON_TYPE and prop_type not in pyleecan_type: 

256 pyleecan_type.append(prop_type) 

257 # Default value as pyleecan type 

258 if ( 

259 type(prop["value"]) is str 

260 and "()" in prop["value"] 

261 and prop["value"][:-2] not in pyleecan_type 

262 ): 

263 pyleecan_type.append(prop["value"][:-2]) 

264 return pyleecan_type
