11def build_geometry(self, alpha=0, delta=0, is_simplified=False): 

12 """Compute the curve (Segment) needed to plot the Slot. 

13 The ending point of a curve is the starting point of the next curve in 

14 the list 

15 

16 Parameters 

17 ---------- 

18 self : HoleM57 

19 A HoleM57 object 

20 alpha : float 

21 Angle to rotate the slot (Default value = 0) [rad] 

22 delta : complex 

23 Complex to translate the slot (Default value = 0) 

24 is_simplified : bool 

25 True to avoid line superposition 

26 

27 Returns 

28 ------- 

29 surf_list: list 

30 List of SurfLine needed to draw the HoleM57 

31 

32 """ 

33 

34 if self.get_is_stator(): # check if the slot is on the stator 

35 st = "S" 

36 else: 

37 st = "R" 

38 Rbo = self.get_Rbo() 

39 

40 # "Tooth" angle (P1',0,P1) 

41 alpha_T = 2 * arcsin(self.W3 / (2 * (Rbo - self.H1))) 

42 # magnet pole pitch angle (Z1,0,Z1') 

43 alpha_S = (2 * pi / self.Zh) - alpha_T 

44 # Angle (P1,P1',P4') and (P5',P4', ) 

45 alpha = (pi - self.W0) / 2 

46 # Half slot pitch 

47 hssp = pi / self.Zh 

48 

49 Z1 = (Rbo - self.H1) * exp(-1j * alpha_S / 2) 

50 x11 = 2 * sin(alpha_S / 2) * (Rbo - self.H1) # Distance from P1 to P1' 

51 # In rect triangle P4, P1, perp (P1,P1') with P4 

52 H = tan(alpha) * (x11 / 2 - self.W1 / 2) 

53 Z4 = Z1.real - H - 1j * self.W1 / 2 

54 

55 x45 = self.H2 / cos(alpha) # distance from P4 to P5 

56 Z5 = Z4 - x45 

57 

58 # Get coordinates of "random" points on (P5,P8) and (P1,P8) 

59 # In ref P4 center and P1 on X+ axis 

60 Z58 = (self.W4 - 1j * self.H2) * exp(1j * angle(Z1 - Z4)) + Z4 

61 # In the tooth ref 

62 Z18 = (Rbo - self.H1 - self.H2 + 1j * self.W3 / 2) * exp(-1j * hssp) 

63 Z8 = inter_line_line(Z5, Z58, Z1, Z18)[0] 

64 

65 # In ref "b" P4 center and P1 on X+ axis 

66 Z8b = (Z8 - Z4) * exp(-1j * angle(Z1 - Z4)) 

67 Z2 = (Z8b + 1j * self.H2 - self.W2) * exp(1j * angle(Z1 - Z4)) + Z4 

68 Z3 = (Z8b + 1j * self.H2 - self.W2 - self.W4) * exp(1j * angle(Z1 - Z4)) + Z4 

69 Z7 = (Z8b - self.W2) * exp(1j * angle(Z1 - Z4)) + Z4 

70 Z6 = (Z8b - self.W2 - self.W4) * exp(1j * angle(Z1 - Z4)) + Z4 

71 

72 # Symmetry 

73 Z1s = Z1.conjugate() 

74 Z2s = Z2.conjugate() 

75 Z3s = Z3.conjugate() 

76 Z4s = Z4.conjugate() 

77 Z5s = Z5.conjugate() 

78 Z6s = Z6.conjugate() 

79 Z7s = Z7.conjugate() 

80 Z8s = Z8.conjugate() 

81 

82 surf_list = list()

98 curve_list_air = list()
